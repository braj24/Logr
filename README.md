# Logreglib
for Login and Registration

# Download

Add the dependencies

    dependencies {
    compile 'com.github.braj24:testlib:1.0.2'
    }
    
# Usage
For login
Login login=new Login(context,"username","password");
login.runLogin();

For Registration

Register register =new Register(context,"username","emailid","password");
runRegister();
